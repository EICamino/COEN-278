get '/' do
    @title = "home"
    erb :home, :layout=>:homelayout
end

get '/about' do
    @title = "about"
    erb :about
end

get '/contact' do
    @title = "contact"
    erb :contact    # actually => erb(:contact)
end

get '/*' do
    @route = params[:splat]
    pass    # go on to not_found
end

not_found do
    erb :contact, :layout=>false
end
