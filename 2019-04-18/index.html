    class VowelFinder
    include Enumerable
    def initialize(string)
        @string = string
    end

    def each
        @string.scan(/[aeiou]/) do |vowel|
            yield vowel
        end
    end
end

"about" # a, o, u

vf = VowelFinder.new("For a mixin, this means you can call them.")
vf.each {|v| p v, '-'}

p vf.select {|v| v >= 'o'}
p vf.map {|v| v*5}
p vf.inject {:+}    # concatenate all
